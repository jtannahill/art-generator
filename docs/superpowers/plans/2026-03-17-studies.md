**Goal:** Add weather event tracking ("Studies") that auto-detects persistent/clustered weather patterns and presents them as multi-day narrative timelines with animated Mapbox paths and artwork grids.

**Architecture:** Study Detector Lambda runs before site rebuild in the Step Function, scanning recent weather data for persistent (3+ consecutive days) and clustered (3+ nearby points) events. Studies link existing artwork — no new Bedrock generation. Study Admin Lambda provides management via Function URL. Site rebuild generates study index + detail pages.

**Tech Stack:** Python 3.12 Lambda, DynamoDB (existing table), Mapbox GL JS, Jinja2, CDK (TypeScript)

"""Study Detector Lambda — auto-detects weather events and updates active studies."""
import json
import os
from collections import defaultdict
from datetime import datetime, timedelta, timezone
from decimal import Decimal

import boto3

TABLE_NAME = os.environ.get("TABLE_NAME", "art-generator")
CELL_SIZE = 5  # degrees for grid grouping
MIN_PERSISTENT_DAYS = 3
MIN_CLUSTER_POINTS = 3
CLUSTER_MAX_DISTANCE = 15  # degrees
STUDY_DURATION_DAYS = 10


def handler(event, context):
    """Scans recent weather data, detects persistent/cluster events, updates studies."""
    dynamodb = boto3.resource("dynamodb")
    table = dynamodb.Table(TABLE_NAME)

    # Get all items
    items = scan_all(table)
    weather_items = [i for i in items if i.get("PK", "").startswith("WEATHER#")]
    study_items = [i for i in items if i.get("PK", "").startswith("STUDY#")]

    # Filter to last 10 days
    cutoff = (datetime.now(timezone.utc) - timedelta(days=10)).strftime("%Y-%m-%d")
    recent = [w for w in weather_items if w.get("date", "") >= cutoff]

    # Get existing studies
    existing_studies = [s for s in study_items if s.get("SK") == "META"]

    # Group by grid
    grid = group_by_grid(recent, CELL_SIZE)

    # Detect events
    persistent = detect_persistent(grid, MIN_PERSISTENT_DAYS)
    clusters = detect_clusters(grid, MIN_CLUSTER_POINTS, CLUSTER_MAX_DISTANCE)

    # Create suggested studies (with duplicate check)
    created = 0
    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    for p in persistent:
        if not is_duplicate_study(existing_studies, p["center_lat"], p["center_lng"],
                                   p["dates"][0], p["dates"][-1]):
            study_id = f"persistent-{p['center_lat']:.0f}-{p['center_lng']:.0f}-{p['dates'][0]}"
            end = (datetime.strptime(p["dates"][0], "%Y-%m-%d") + timedelta(days=STUDY_DURATION_DAYS)).strftime("%Y-%m-%d")
            table.put_item(Item={
                "PK": f"STUDY#{study_id}",
                "SK": "META",
                "study_id": study_id,
                "name": f"Persistent Event {p['center_lat']:.0f}°{'N' if p['center_lat'] >= 0 else 'S'}, {abs(p['center_lng']):.0f}°{'E' if p['center_lng'] >= 0 else 'W'}",
                "status": "suggested",
                "artist": None,
                "start_date": p["dates"][0],
                "end_date": end,
                "detection_type": "persistent",
                "coordinates": [{"lat": int(p["center_lat"]), "lng": int(p["center_lng"]), "label": "center"}],
                "created_at": datetime.now(timezone.utc).isoformat(),
                "updated_at": datetime.now(timezone.utc).isoformat(),
            })
            created += 1

    for c in clusters:
        center_lat = sum(co["lat"] for co in c["coordinates"]) / len(c["coordinates"])
        center_lng = sum(co["lng"] for co in c["coordinates"]) / len(c["coordinates"])
        if not is_duplicate_study(existing_studies, center_lat, center_lng, c["date"], c["date"]):
            study_id = f"cluster-{center_lat:.0f}-{center_lng:.0f}-{c['date']}"
            end = (datetime.strptime(c["date"], "%Y-%m-%d") + timedelta(days=STUDY_DURATION_DAYS)).strftime("%Y-%m-%d")
            table.put_item(Item={
                "PK": f"STUDY#{study_id}",
                "SK": "META",
                "study_id": study_id,
                "name": f"Regional Event {center_lat:.0f}°{'N' if center_lat >= 0 else 'S'}, {abs(center_lng):.0f}°{'E' if center_lng >= 0 else 'W'}",
                "status": "suggested",
                "artist": None,
                "start_date": c["date"],
                "end_date": end,
                "detection_type": "cluster",
                "coordinates": c["coordinates"],
                "created_at": datetime.now(timezone.utc).isoformat(),
                "updated_at": datetime.now(timezone.utc).isoformat(),
            })
            created += 1

    # Update active studies with today's matching artworks
    active_studies = [s for s in existing_studies if s.get("status") == "active"]
    updated = 0
    for study in active_studies:
        coords = study.get("coordinates", [])
        if not coords:
            continue

        # Find today's artworks matching study coordinates
        today_artworks = [w for w in recent if w.get("date") == today]
        matching = []
        for art in today_artworks:
            art_lat = float(art.get("lat", 0))
            art_lng = float(art.get("lng", 0))
            for coord in coords:
                if abs(art_lat - float(coord["lat"])) <= 5 and abs(art_lng - float(coord["lng"])) <= 5:
                    matching.append({
                        "run_id": art.get("run_id", art.get("PK", "").replace("WEATHER#", "")),
                        "slug": art.get("SK", ""),
                        "lat": art_lat,
                        "lng": art_lng,
                    })
                    break

        if matching:
            # Compute weather summary
            temps = [float(a.get("temp", 0)) for a in today_artworks if any(
                abs(float(a.get("lat", 0)) - float(c["lat"])) <= 5 for c in coords)]
            winds = [float(a.get("wind_speed", 0)) for a in today_artworks if any(
                abs(float(a.get("lat", 0)) - float(c["lat"])) <= 5 for c in coords)]
            pressures = [float(a.get("pressure", 0)) for a in today_artworks if any(
                abs(float(a.get("lat", 0)) - float(c["lat"])) <= 5 for c in coords)]

            table.put_item(Item={
                "PK": study["PK"],
                "SK": f"DAY#{today}",
                "date": today,
                "artwork_refs": matching,
                "weather_summary": {
                    "avg_pressure": Decimal(str(round(sum(pressures) / len(pressures), 1))) if pressures else None,
                    "max_wind": Decimal(str(round(max(winds), 1))) if winds else None,
                    "min_temp": Decimal(str(round(min(temps), 1))) if temps else None,
                    "max_temp": Decimal(str(round(max(temps), 1))) if temps else None,
                },
            })
            updated += 1

        # Auto-complete if past end date
        if study.get("end_date", "") < today:
            table.update_item(
                Key={"PK": study["PK"], "SK": "META"},
                UpdateExpression="SET #s = :s, updated_at = :u",
                ExpressionAttributeNames={"#s": "status"},
                ExpressionAttributeValues={":s": "complete", ":u": datetime.now(timezone.utc).isoformat()},
            )

    return {"created": created, "updated": updated}


def scan_all(table):
    """Full DynamoDB scan with pagination."""
    items = []
    params = {}
    while True:
        response = table.scan(**params)
        items.extend(response.get("Items", []))
        if "LastEvaluatedKey" not in response:
            break
        params["ExclusiveStartKey"] = response["LastEvaluatedKey"]
    return items


def group_by_grid(items, cell_size=5):
    """Group weather items into grid cells of cell_size degrees."""
    grid = defaultdict(list)
    for item in items:
        lat = float(item.get("lat", 0))
        lng = float(item.get("lng", 0))
        cell = (int(lat // cell_size) * cell_size, int(lng // cell_size) * cell_size)
        grid[cell].append(item)
    return dict(grid)


def detect_persistent(grid, min_days=3):
    """Find grid cells with items on 3+ consecutive days."""
    suggestions = []
    for (cell_lat, cell_lng), items in grid.items():
        dates = sorted(set(item.get("date", "")[:10] for item in items))
        if len(dates) < min_days:
            continue

        # Find consecutive runs
        consecutive = 1
        best_run = [dates[0]]
        current_run = [dates[0]]

        for i in range(1, len(dates)):
            prev = datetime.strptime(dates[i - 1], "%Y-%m-%d")
            curr = datetime.strptime(dates[i], "%Y-%m-%d")
            if (curr - prev).days == 1:
                current_run.append(dates[i])
            else:
                if len(current_run) > len(best_run):
                    best_run = current_run
                current_run = [dates[i]]

        if len(current_run) > len(best_run):
            best_run = current_run

        if len(best_run) >= min_days:
            suggestions.append({
                "center_lat": cell_lat,
                "center_lng": cell_lng,
                "dates": best_run,
                "items": [i for i in items if i.get("date", "")[:10] in best_run],
            })

    return suggestions


def detect_clusters(grid, min_points=3, max_distance=15):
    """Find days where 3+ grid cells within max_distance are all active."""
    # Group cells by date
    date_cells = defaultdict(list)
    for (cell_lat, cell_lng), items in grid.items():
        for item in items:
            date = item.get("date", "")[:10]
            date_cells[date].append({
                "lat": cell_lat,
                "lng": cell_lng,
                "item": item,
            })

    suggestions = []
    for date, cells in date_cells.items():
        if len(cells) < min_points:
            continue

        # Simple clustering: find groups of cells within max_distance
        used = set()
        for i, c1 in enumerate(cells):
            if i in used:
                continue
            cluster = [c1]
            for j, c2 in enumerate(cells):
                if j <= i or j in used:
                    continue
                if abs(c1["lat"] - c2["lat"]) <= max_distance and abs(c1["lng"] - c2["lng"]) <= max_distance:
                    cluster.append(c2)

            if len(cluster) >= min_points:
                for c in cluster:
                    used.add(cells.index(c))
                suggestions.append({
                    "date": date,
                    "coordinates": [{"lat": c["lat"], "lng": c["lng"]} for c in cluster],
                })

    return suggestions


def is_duplicate_study(existing_studies, lat, lng, start_date, end_date):
    """Check if a study already covers this region + time range."""
    for study in existing_studies:
        coords = study.get("coordinates", [])
        if not coords:
            continue
        # Check region overlap (center within 10 degrees)
        for coord in coords:
            if abs(float(coord.get("lat", 0)) - lat) <= 10 and abs(float(coord.get("lng", 0)) - lng) <= 10:
                # Check date overlap
                s_start = study.get("start_date", "")
                s_end = study.get("end_date", "")
                if s_start <= end_date and s_end >= start_date:
                    return True
    return False

"""Study Admin Lambda — CRUD for studies via Function URL."""
import json
import os
import re
from datetime import datetime, timezone

import boto3

TABLE_NAME = os.environ.get("TABLE_NAME", "art-generator")
ADMIN_API_KEY = os.environ.get("ADMIN_API_KEY", "")


def handler(event, context):
    """Lambda Function URL handler for study management."""
    # Auth check
    auth = event.get("headers", {}).get("authorization", "")
    if ADMIN_API_KEY and auth != ADMIN_API_KEY:
        return {"statusCode": 401, "headers": {"Content-Type": "application/json"},
                "body": json.dumps({"error": "Unauthorized"})}

    query = event.get("queryStringParameters") or {}
    action = query.get("action", "list")

    dynamodb = boto3.resource("dynamodb")
    table = dynamodb.Table(TABLE_NAME)

    if action == "list":
        return _list_studies(table)
    elif action == "create":
        return _create_study(table, query)
    elif action == "approve":
        return _update_status(table, query.get("id", ""), "active")
    elif action == "complete":
        return _update_status(table, query.get("id", ""), "complete")
    elif action == "delete":
        return _delete_study(table, query.get("id", ""))
    else:
        return {"statusCode": 400, "headers": {"Content-Type": "application/json"},
                "body": json.dumps({"error": f"Unknown action: {action}"})}


def _list_studies(table):
    items = []
    params = {"FilterExpression": "begins_with(PK, :p) AND SK = :meta",
              "ExpressionAttributeValues": {":p": "STUDY#", ":meta": "META"}}
    while True:
        resp = table.scan(**params)
        items.extend(resp.get("Items", []))
        if "LastEvaluatedKey" not in resp:
            break
        params["ExclusiveStartKey"] = resp["LastEvaluatedKey"]

    return {"statusCode": 200, "headers": {"Content-Type": "application/json"},
            "body": json.dumps(items, default=str)}


def _create_study(table, params):
    parsed = parse_create_params(params)
    study_id = generate_study_id(parsed["name"], parsed["start_date"])

    table.put_item(Item={
        "PK": f"STUDY#{study_id}",
        "SK": "META",
        "study_id": study_id,
        "name": parsed["name"],
        "status": "active",
        "artist": parsed.get("artist"),
        "start_date": parsed["start_date"],
        "end_date": parsed["end_date"],
        "detection_type": "manual",
        "coordinates": parsed["coordinates"],
        "created_at": datetime.now(timezone.utc).isoformat(),
        "updated_at": datetime.now(timezone.utc).isoformat(),
    })

    return {"statusCode": 200, "headers": {"Content-Type": "application/json"},
            "body": json.dumps({"status": "created", "study_id": study_id})}


def _update_status(table, study_id, new_status):
    if not study_id:
        return {"statusCode": 400, "headers": {"Content-Type": "application/json"},
                "body": json.dumps({"error": "Missing id"})}

    table.update_item(
        Key={"PK": f"STUDY#{study_id}", "SK": "META"},
        UpdateExpression="SET #s = :s, updated_at = :u",
        ExpressionAttributeNames={"#s": "status"},
        ExpressionAttributeValues={":s": new_status, ":u": datetime.now(timezone.utc).isoformat()},
    )

    return {"statusCode": 200, "headers": {"Content-Type": "application/json"},
            "body": json.dumps({"status": "updated", "study_id": study_id, "new_status": new_status})}


def _delete_study(table, study_id):
    if not study_id:
        return {"statusCode": 400, "headers": {"Content-Type": "application/json"},
                "body": json.dumps({"error": "Missing id"})}

    # Delete META and all DAY# entries
    resp = table.query(KeyConditionExpression="PK = :pk",
                       ExpressionAttributeValues={":pk": f"STUDY#{study_id}"})
    for item in resp.get("Items", []):
        table.delete_item(Key={"PK": item["PK"], "SK": item["SK"]})

    return {"statusCode": 200, "headers": {"Content-Type": "application/json"},
            "body": json.dumps({"status": "deleted", "study_id": study_id})}


def parse_create_params(params):
    """Extract study creation fields from query params."""
    lats = params.get("lat", "0").split(",")
    lngs = params.get("lng", "0").split(",")
    coordinates = [{"lat": float(lat), "lng": float(lng)} for lat, lng in zip(lats, lngs)]

    return {
        "name": params.get("name", "Untitled Study"),
        "start_date": params.get("start", datetime.now(timezone.utc).strftime("%Y-%m-%d")),
        "end_date": params.get("end", (datetime.now(timezone.utc) + __import__("datetime").timedelta(days=10)).strftime("%Y-%m-%d")),
        "artist": params.get("artist"),
        "coordinates": coordinates,
    }


def generate_study_id(name, start_date):
    """Create a slug from name + date."""
    slug = re.sub(r"[^a-z0-9]+", "-", name.lower()).strip("-")
    return f"{slug}-{start_date}"
